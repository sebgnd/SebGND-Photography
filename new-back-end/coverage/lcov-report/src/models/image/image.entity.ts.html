import { Entity, Column, ManyToOne } from 'typeorm';
import { Table } from '../../common/helpers/database';
 
import { Category } from '../category/category.entity';
import { Camera } from '../camera/camera.entity';
import { Lense } from '../lense/lense.entity';
 
@Entity()
export class Image extends Table {
	@Column({ nullable: true })
	iso: number;
 
	@Column({ nullable: true })
	shutterSpeed: string;
 
	@Column({ nullable: true })
	aperture: string;
 
	@Column({ nullable: true })
	focalLength: string;
 
	@Column()
	height: number;
 
	@Column()
	width: number;
 
	@Column()
	isCategoryThumbnail: boolean;
 
	@ManyToOne(() => Category, (category) => category.images)
	category: Category;
 
	@ManyToOne(() => Camera, (camera) => camera.images)
	camera: Camera;
 
	@ManyToOne(() => Lense, (lense) => lense.images)
	lense: Lense;
}
